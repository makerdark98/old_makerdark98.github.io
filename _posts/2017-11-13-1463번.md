## 1463번 1로 만들기
* BFS를 사용하여 찾아주면 된다.
* 같은곳을 들어가지 않도록 check를 하면서 하면 된다.

## Source 1463.cpp
```
#include <iostream>
#include <queue>
#define sci(n) scanf("%d",&(n))
using namespace std;
typedef pair<int, int> pii;
int chk[1000001];
int main(){
    queue<pii> q;
    int n;
    sci(n);
    q.push({n,0});
    while(!q.empty()){
        auto f = q.front();
        q.pop();
        if(f.first==1){
            printf("%d\n",f.second);
            return 0;
        }
        if(chk[f.first]==0){
            chk[f.first]=f.second;
            if(f.first%3==0) q.push({f.first/3, f.second+1});
            if(f.first%2==0) q.push({f.first/2, f.second+1});
            q.push({f.first-1, f.second+1});
        }
    }
    return 0;
}
```
